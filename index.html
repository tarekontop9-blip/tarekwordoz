async function joinLobby() { name = document.getElementById("username").value; lobby = document.getElementById("code").value.toUpperCase(); if (!name) { alert("Scrie username!"); return; } const res = await fetch(`/api/lobby?action=join&code=${lobby}&name=${name}`); const data = await res.json(); if (data.error) { document.getElementById("status").innerText = data.error; return; } document.getElementById("status").innerText = "Lobby: " + lobby; startPolling(); }
